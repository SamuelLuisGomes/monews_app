      appBar: MenuVoltar(),

          child: ListView.builder(
        itemBuilder: (context, index) {
          if (!_isLoading) {
            return index == 0
                ? MySearch(
                    hintText: 'Petrobras ......',
                    onChanged: (searchText) {
                      searchText = searchText.toLowerCase();
                      setState(() {
                        return atualizaPesquisa(searchText);
                      });
                    },
                  )
                : MyListNoticias(
                    noticias: _noticiasDisplay[index],
                  );
          } else {
            return const MyLoading();
          }
        },
        itemCount: _noticiasDisplay.length,

  void atualizaPesquisa(String searchText) {
    setState(() {
      //_isLoading = true;
      txt = searchText;
    });
    fetchNoticiasPesquisa(txt).then((value) {
      setState(() {
        _isLoading = false;
        _noticias.addAll(value);
        _noticiasDisplay = _noticias;
      });
    });
